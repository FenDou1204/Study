<script language="javascript" type="text/javascript">
    var DEPARTURE = "#ctl00_MainContentPlaceHolder_ucSearch_lblDepartureStation";
    var ARRIVAL = "#ctl00_MainContentPlaceHolder_ucSearch_lblArrivalStation";
    var NUMBER = "#ctl00_MainContentPlaceHolder_ucSearch_lblNumber";
    var STATION = "#ctl00_MainContentPlaceHolder_ucSearch_lblStation";
    $.ready(function () {
        ChangeCatalog(2);
    });

    function Search() {
        var validateShow;
        $.mod.load('validate', '1.1', function() {
            var valid = $(document).regMod("validate", "1.1");
            validateShow = function(obj, message) {
                valid.method("show", { $obj: obj, data: message, removeErrorClass: true, hideEvent: "blur", isFocus: true });
            };
        });
        var baseUrl = "";

        if ($("#ctl00_MainContentPlaceHolder_ucSearch_rbStation2Station")[0].checked) {
            if ($("#notice01")[0].value.trim() == "中文/拼音/首字母") {
                validateShow($("#notice01"), "请输入出发城市名称");
                return false;
            }
            if ($("#notice02")[0].value.trim() == "中文/拼音/首字母") {
                validateShow($("#notice02"), "请输入到达站名称");
                return false;
            }
            baseUrl = "http://trains.ctrip.com/TrainBooking" + "/TrainSchedule/Station2Station.aspx?from=" + $("#from")[0].value.trim() + "&to=" + $("#to")[0].value.trim() + "&fromCn=" + $("#notice01")[0].value.trim() + "&toCn=" + $("#notice02")[0].value.trim();
        } else if ($("#ctl00_MainContentPlaceHolder_ucSearch_rbTrainNumber")[0].checked) {
            var reg = /^[A-Za-z]?\d+$/ig;
            if (!reg.test($("#tn")[0].value.trim())) {
                validateShow($("#tn"), "请输入有效车次名");
                return false;
            }
            baseUrl = "http://trains.ctrip.com/TrainSchedule" + "/" + $("#tn")[0].value.trim();
        } else if ($("#ctl00_MainContentPlaceHolder_ucSearch_rbStation")[0].checked) {
            if ($("#notice03")[0].value.trim() == "中文/拼音/首字母"||$("#notice03")[0].value.trim().length<=0) {
                validateShow($("#notice03"), "请输入车站名称");
                return false;
            }
            baseUrl = "http://trains.ctrip.com/TrainBooking" + "/TrainSchedule/Station.aspx?station=" + $("#station")[0].value.trim() + "&stationCn=" + $("#notice03")[0].value.trim();
        }
        if (baseUrl.trim().length != 0) {
            location.href = baseUrl.trim();
        }
        return false;
    }

    function ChangeCatalog(catalog) {
        $(NUMBER).css("display", "none");
        $(STATION).css("display", "none");
        $(DEPARTURE).css("display", "none");
        $(ARRIVAL).css("display", "none");
        switch(catalog)
        {        
        case 2:
            $("#ctl00_MainContentPlaceHolder_ucSearch_rbTrainNumber")[0].checked = true;
            $("searchBox").parentNode.className = "searchbox_seo searchbox_2 layoutfix";
            $(NUMBER).css("display", "");
            break;
        case 3:
            $("#ctl00_MainContentPlaceHolder_ucSearch_rbStation")[0].checked = true;
            $("searchBox").parentNode.className = "searchbox_seo searchbox_3 layoutfix";
            $(STATION).css("display", "");
            break;
        default:
            $("#ctl00_MainContentPlaceHolder_ucSearch_rbStation2Station")[0].checked = true;
            $("searchBox").parentNode.className = "searchbox_seo searchbox_1 layoutfix";
            $(DEPARTURE).css("display", "");
            $(ARRIVAL).css("display", "");
            break;
        }
    }
